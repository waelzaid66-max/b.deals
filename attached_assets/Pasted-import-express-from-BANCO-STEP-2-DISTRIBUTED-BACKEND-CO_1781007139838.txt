import { PrismaClient } from "@prisma/client";

function validateAttributes(category: string, attrs: any) {
  if (!attrs || typeof attrs !== "object") {
    throw new Error("Invalid attributes");
  }

  if (category === "car") {
    if (!attrs.mileage || !attrs.condition) {
      throw new Error("Car must include mileage & condition");
    }
  }

  if (category === "industrial") {
    if (!attrs.capacity) {
      throw new Error("Industrial asset must include capacity");
    }
  }

  return true;
}

/**
 * PAYMENT NORMALIZATION
 */
function normalizePaymentOptions(options: any[]) {
  if (!options || options.length === 0) {
    return { has_installment: false };
  }

  const installments = options.filter(
    (o) => o.monthly_installment && o.monthly_installment > 0
  );

  if (installments.length === 0) {
    return { has_installment: false };
  }

  const lowestMonthly = installments.reduce((min, curr) =>
    curr.monthly_installment < min.monthly_installment ? curr : min
  );

  const lowestDown = installments.reduce((min, curr) =>
    curr.down_payment < min.down_payment ? curr : min
  );

  return {
    has_installment: true,
    lowest_monthly: lowestMonthly.monthly_installment,
    lowest_down_payment: lowestDown.down_payment,
    badge_text: `Starts from ${lowestMonthly.monthly_installment}/month`,
  };
}

/**
 * RANKING ENGINE
 */
function rankListing(listing: any) {
  const now = Date.now();
  const created = new Date(listing.created_at).getTime();

  const freshness = Math.max(0, 1 - (now - created) / (1000 * 60 * 60 * 24 * 30));

  const engagement =
    (listing.interactions?.views || 0) +
    (listing.interactions?.clicks || 0);

  const mediaScore = listing.has_video ? 1 : 0;

  const verified = listing.seller?.verified ? 1 : 0;

  const score =
    freshness * 0.4 +
    engagement * 0.3 +
    mediaScore * 0.2 +
    verified * 0.1;

  return score;
}

app.post("/api/listings/create", async (req, res) => {
  const { listing, attributes, media, payments, location } = req.body;

  try {
    const result = await prisma.$transaction(async (tx) => {
      validateAttributes(listing.category, attributes);

      const newListing = await tx.listings.create({
        data: {
          user_id: listing.user_id,
          title: listing.title,
          category: listing.category,
          price: listing.price,
          location: listing.location,
          status: "active",
        },
      });

      await tx.listing_attributes.create({
        data: {
          listing_id: newListing.id,
          attributes,
        },
      });

      await tx.listing_locations.create({
        data: {
          listing_id: newListing.id,
          lat: location.lat,
          lng: location.lng,
        },
      });

      if (!media || media.length === 0) {
        throw new Error("Media required");
      }

      await tx.listing_media.createMany({
        data: media.map((m: any) => ({
          listing_id: newListing.id,
          type: m.type,
          url: m.url,
          is_primary: m.is_primary || false,
        })),
      });

      if (payments && payments.length > 0) {
        await tx.payment_options.createMany({
          data: payments.map((p: any) => ({
            listing_id: newListing.id,
            type: p.type,
            down_payment: p.down_payment,
            monthly_payment: p.monthly_payment,
            duration: p.duration,
          })),
        });
      }

      return newListing;
    });

    res.json({ success: true, listing: result });
  } catch (error: any) {
    res.status(400).json({ error: error.message });
  }
});

app.get("/api/feed", async (req, res) => {
  const page = Number(req.query.page || 1);
  const limit = 20;
  const offset = (page - 1) * limit;

  try {
    const listings = await prisma.listings.findMany({
      skip: offset,
      take: limit,
      include: {
        listing_attributes: true,
        listing_media: true,
        payment_options: true,
        interactions: true,
        user: true,
      },
    });

    const enriched = listings.map((l: any) => {
      const hasVideo = l.listing_media.some((m: any) => m.type === "video");

      const normalizedPayment = normalizePaymentOptions(
        l.payment_options
      );

      return {
        ...l,
        has_video: hasVideo,
        payment_summary: normalizedPayment,
        score: rankListing({
          ...l,
          has_video: hasVideo,
          seller: l.user,
        }),
      };
    });

    const sorted = enriched.sort((a, b) => b.score - a.score);

    // inject ads
    const finalFeed = [];
    for (let i = 0; i < sorted.length; i++) {
      if (i > 0 && i % 6 === 0) {
        finalFeed.push({ type: "ad", message: "Sponsored Listing" });
      }
      finalFeed.push(sorted[i]);
    }

    res.json(finalFeed);
  } catch (error: any) {
    res.status(500).json({ error: error.message });
  }
});

      data: {
        clicks: { increment: 1 },
      },
    });

    res.json({ success: true });
  } catch (error: any) {
    res.status(400).json({ error: error.message });
  }
